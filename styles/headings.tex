\usepackage{fancyhdr}

% headings is a default style for sections and subsections
\fancypagestyle{headings}{
    \renewcommand{\headrulewidth}{0pt}
    \fancyhf{}
    \fancyhead[R]{\thepage}
}

% plain is a default style for chapters
\fancypagestyle{plain}{
    \renewcommand{\headrulewidth}{0pt}
    \fancyhf{}
    \fancyhead[R]{\thepage}
}
